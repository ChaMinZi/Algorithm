class Solution {
    fun solution(clothes: Array<Array<String>>): Int {
        var answer = 1

        val maps = mutableMapOf<String, Int>()

        for (clothe in clothes) {
            val key = clothe[1]
            if (maps.containsKey(key)) {
                maps[key] = maps[key]!! + 1
            } else {
                maps[key] = 1
            }
        }

        for (m in maps) {
            answer *= (m.value + 1)
        }

        return answer - 1
    }
}
