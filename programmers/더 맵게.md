힙 짜는 문제였습니다.

## C++
```c++
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

struct priority_queue {
	int buffer[1000002];

	void init(int size) {
		cur = 0;
		_size = 0;
	}

	int cur, _next, _size, par;
	
	int getSize() {
		return _size;
	}

	int top() {
		return buffer[0];
	}

	void swap(int a, int b) {
		int temp = buffer[a];
		buffer[a] = buffer[b];
		buffer[b] = temp;
	}

	void push(int value) {
		buffer[_size++] = value;
		cur = _size - 1;

		while (cur != 0) {
			par = (cur - 1) / 2;
			if (buffer[par] > buffer[cur])
				swap(par, cur);
			else
				break;
			cur = par;
		}
	}

	void pop() {
		buffer[0] = buffer[_size - 1];
		_size--;
		cur = 0;
		
		while (cur * 2 + 1 < _size) {
			_next = cur;

			if (buffer[cur * 2 + 1] < buffer[_next]) _next = cur * 2 + 1;
			if (cur * 2 + 2 < _size && buffer[cur * 2 + 2] < buffer[_next]) _next = cur * 2 + 2;
			
			if (cur == _next)
				break;

			swap(cur, _next);
			cur = _next;
		}
	}
};

int solution(vector<int> scoville, int K) {
	int answer = 0;
	priority_queue pq;

	bool flag = true;

	// 힙 만들기
	pq.init(scoville.size());
	for (int i = 0; i < scoville.size(); i++) {
		pq.push(scoville[i]);

		if (scoville[i] < K)
			flag = false;
	}

	// 모두 다 K 이상
	if (flag)
		return answer;

	int t1, t2;
	while (pq._size > 1) {
		t1 = pq.top();
		pq.pop();
		t2 = pq.top();
		pq.pop();

		pq.push(t1 + (t2 * 2));
		
		++answer;
		if (pq.top() >= K)
			return answer;
	}

	if (pq.top() >= K)
		return answer;
	else
		return -1;
}
```
