```c++
#include <iostream>
#include <string>
#include <vector>
#include <string.h>
#include <sstream>
#include <algorithm>
#include <map>
#include <set>
#include <queue>

using namespace std;

int N;
int score, now = 1;
int dr[4] = { 0, 0, 0, 1 };
int dc[4] = { 0, 0, 1, 0 };

vector<vector<int>> g(6, vector<int>(4, 0)), b(6, vector<int>(4, 0));

int status(vector<int>& x) {
    int cnt0 = 0;
    for (int i = 0; i < 4; i++) {
        if (x[i] == 0) cnt0++;
    }
    if (cnt0 == 4) return 0;
    if (cnt0 == 0) return 1;
    return 2;
}

void addBlock(int t, int col, vector<vector<int>>& board, int row) {
    while (1) {
        if (board[row][col] || board[row + dr[t]][col + dc[t]]) {
            row--;
            break;
        }
        if (t == 3 && row == 4)
            break;
        if (row == 5)
            break;
        row++;
    }
    board[row][col] = board[row + dr[t]][col + dc[t]] = now++;
}

void add(int t, int col, vector<vector<int>>& board) {
    addBlock(t, col, board, 0);

    // 행이 꽉 찬 경우
    while (true) {
        int idx = 2;

        while (idx < 6 && status(board[idx]) != 1)
            idx++;
        if (idx == 6)
            break;

        score++;
        board[idx] = vector<int>(4);
        for (int i = idx - 1; i >= 0; i--) {
            for (int j = 0; j < 4; j++) {
                if (board[i][j] == 0) continue;
                if (j != 3 && board[i][j] == board[i][j + 1]) { // 2번 블럭
                    board[i][j] = board[i][j + 1] = 0;
                    addBlock(2, j, board, i);
                }
                else if (i != 0 && board[i - 1][j] == board[i][j]) { // 3번 블럭
                    board[i - 1][j] = board[i][j] = 0;
                    addBlock(3, j, board, i - 1);
                }
                else { // 1번 블럭 형태
                    board[i][j] = 0;
                    addBlock(1, j, board, i);
                }
            }
        }
    }

    // 0, 1 행에 블록이 있는 경우
    if (status(board[0]) != 0) { // 0행이 비지 않은 경우(1행도 당연히 비지 않았음)
        board.pop_back();
        board.pop_back();
        board.insert(board.begin(), 2, vector<int>(4));
    }
    else if (status(board[1]) != 0) { // 0행은 비었는데 1행은 비지 않은 경우
        board.pop_back();
        board.insert(board.begin(), vector<int>(4));
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    // inin
    score = 0;
    now = 1;

    int t, r, c;
    cin >> N;
    while (N--) {
        now++;

        cin >> t >> r >> c;
        if (t == 1) {
            add(1, c, g);
            add(1, r, b);
        }
        else if (t == 2) {
            add(2, c, g);
            add(3, r, b);
        }
        else if (t == 3) {
            add(3, c, g);
            add(2, r, b);
        }
    }

    int cnt = 0;
    for (int i = 0; i < 6; i++) {
        for (int j = 0; j < 4; j++)
            cnt += (b[i][j] != 0) + (g[i][j] != 0);
    }
    cout << score << '\n' << cnt;

    return 0;
}
```
