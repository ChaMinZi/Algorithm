## 풀이 과정
    1. 2개의 구역으로 조합 만듦
    2. 각각의 도시가 모두 연결되어 있는지 확인
    3. 도시가 모두 연결되어 있다면 최소값인지 계산

---
```c++
#include <iostream>
#include <algorithm>
#include <cstring>
#include <queue>
#include <vector>

using namespace std;

int N, res;
bool board[11][11] = { false, };
int people[11];

bool isPossi(vector<int> sets) {
	queue<int> que;
	bool chk[11];
	int cur;
	int cnt;

	for (int i = 0; i < sets.size(); i++) {
		while (!que.empty())
			que.pop();
		memset(chk, false, sizeof(chk));
		cnt = 0;
		
		que.push(sets[i]);
		chk[sets[i]] = true;
		while (!que.empty()) {
			cur = que.front();
			que.pop();
			
			for (int j = 0; j < sets.size(); j++) {
				if (!board[cur][sets[j]] || chk[sets[j]])
					continue;
				
				chk[sets.at(j)] = true;
				que.push(sets[j]);
			}
		}

		for (int j = 0; j < sets.size(); j++) {
			if (chk[sets[j]])
				cnt++;
		}

		if (cnt == sets.size())
			return true;
	}

	return false;
}

void update(vector<int> fset, vector<int> sset) {
	int fsum, ssum;
	fsum = ssum = 0;

	for (int i = 0; i < fset.size(); i++) {
		fsum += people[fset[i]];
	}
	for (int i = 0; i < sset.size(); i++) {
		ssum += people[sset[i]];
	}

	res = min(res, abs(fsum - ssum));
}

void solve(int start, int record) {
	if (start >= N) {
		// 기록
		vector<int> fset, sset;
		bool chk[12] = { false, };

		for (int i = 0; i < N; i++) {
			if (record & (1 << i))
				fset.push_back(i);
			else
				sset.push_back(i);
		}

		if (fset.size() == 1 && isPossi(sset)) {
			update(fset, sset);
		}
		else if (sset.size() == 1 && isPossi(fset)) {
			update(fset, sset);
		}
		else if (fset.size() > 1 && sset.size() > 1 && isPossi(fset) && isPossi(sset)) {
			update(fset, sset);
		} 

		return;
	}

	solve(start + 1, record);
	record |= (1 << start);
	solve(start + 1, record);
}

int main() {
	ios_base::sync_with_stdio(false);

	int cnt, temp;
	res = 0xfffffff;

	cin >> N;
	for (int i = 0; i < N; i++) {
		cin >> people[i];
	}

	for (int i = 0; i < N; i++) {
		cin >> cnt;
		for (int j = 0; j < cnt; j++) {
			cin >> temp;
			board[i][temp - 1] = true;
		}
	}

	// 조합 만들기
	solve(0,0);

	if (res == 0xfffffff)
		cout << "-1";
	else
		cout << res;

	return 0;
}
```
